\begin{align}
P[g(X^n)-E[g(x^n)] \geqslant t] \leqslant exp[\frac{-2t^2}{\Sigma_{1=1}^{n}(i^2)}]
\end{align}
\begin{align}
P[g(x^n)-E[g(x^n)] \leqslant -t] \leqslant exp[\frac{-2t^2}{\Sigma_{1=1}^{n}(i^2)}
